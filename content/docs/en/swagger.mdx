# Swagger / OpenAPI 3.0

AzuraJS includes full support for API documentation using Swagger/OpenAPI 3.0, featuring a modern dark mode interface and built-in **Try It Out** functionality.

> **Note:** AzuraJS Swagger automatically generates OpenAPI 3.0 documentation from your controllers and decorators, with no manual configuration required.

## Installation

Swagger is already included in AzuraJS. Just import it from the `azurajs/swagger` submodule:

```typescript
import { setupSwaggerWithControllers } from "azurajs/swagger";
```

## Quick Setup

The easiest way to add Swagger to your application is with a single setup call:

```typescript
import { AzuraClient } from "azurajs";
import { setupSwaggerWithControllers } from "azurajs/swagger";
import { Controller, Get, Post, Req, Res } from "azurajs/decorators";

@Controller("/users")
class UserController {
  private users = [
    { id: "1", name: "Alice", email: "alice@example.com" },
    { id: "2", name: "Bob", email: "bob@example.com" },
  ];

  @Get("/")
  async listUsers(@Req() req, @Res() res) {
    const page = Number(req.query.page) || 1;
    const limit = Number(req.query.limit) || 10;
    const start = (page - 1) * limit;
    const items = this.users.slice(start, start + limit);
    res.json({ items, total: this.users.length, page });
  }

  @Get("/:id")
  async getUser(@Req() req, @Res() res) {
    const user = this.users.find((u) => u.id === req.params.id);
    if (!user) return res.status(404).json({ error: "User not found" });
    res.json(user);
  }

  @Post("/")
  async createUser(@Req() req, @Res() res) {
    const { name, email } = req.body;
    if (!name || !email) {
      return res.status(400).json({ error: "name and email are required" });
    }
    const user = { id: String(Date.now()), name, email };
    this.users.push(user);
    res.status(201).json(user);
  }
}

const app = new AzuraClient();

setupSwaggerWithControllers(
  app,
  {
    title: "My API",
    description: "API documentation",
    version: "1.0.0",
  },
  [UserController]
);

await app.listen(3000);
```

**Access the documentation at:**

- **Interactive UI:** `http://localhost:3000/docs`
- **OpenAPI JSON:** `http://localhost:3000/api-spec.json`

## Documentation Decorators

### @Swagger() – New Simplified API ⭐

The `@Swagger` decorator centralizes **all** documentation in one place, making your API docs easier to read and maintain:

```typescript
import { AzuraClient } from "azurajs";
import { Controller, Get, Post, Req, Res } from "azurajs/decorators";
import { Swagger, setupSwaggerWithControllers } from "azurajs/swagger";

@Controller("/products")
class ProductController {
  private products = [
    { id: "p1", name: "Laptop", price: 1299.99 },
    { id: "p2", name: "Mouse", price: 29.99 },
  ];

  @Get("/")
  @Swagger({
    summary: "List products",
    description: "Returns a paginated list of products",
    tags: ["Products"],
    parameters: [
      { name: "page", in: "query", schema: { type: "number" }, example: 1 },
      { name: "limit", in: "query", schema: { type: "number" }, example: 10 },
    ],
    responses: {
      200: {
        description: "Product list",
        example: [
          { id: "p1", name: "Laptop", price: 1299.99 },
          { id: "p2", name: "Mouse", price: 29.99 },
        ],
      },
    },
  })
  async list(@Req() req, @Res() res) {
    const page = Number(req.query.page) || 1;
    const limit = Number(req.query.limit) || 10;
    const items = this.products.slice((page - 1) * limit, page * limit);
    res.json(items);
  }

  @Get("/:id")
  @Swagger({
    summary: "Get product by ID",
    description: "Returns detailed information about a product",
    tags: ["Products"],
    parameters: [
      {
        name: "id",
        in: "path",
        description: "Product ID",
        required: true,
        schema: { type: "string" },
        example: "p1",
      },
    ],
    responses: {
      200: {
        description: "Product found",
        example: { id: "p1", name: "Laptop", price: 1299.99 },
      },
      404: {
        description: "Product not found",
        example: { error: "Product not found" },
      },
    },
  })
  async get(@Req() req, @Res() res) {
    const product = this.products.find((p) => p.id === req.params.id);
    if (!product) return res.status(404).json({ error: "Product not found" });
    res.json(product);
  }

  @Post("/")
  @Swagger({
    summary: "Create product",
    tags: ["Products"],
    requestBody: {
      description: "Product data",
      required: true,
      content: {
        "application/json": {
          schema: {
            type: "object",
            properties: {
              name: { type: "string" },
              price: { type: "number" },
            },
            required: ["name", "price"],
          },
          example: { name: "Keyboard", price: 79.99 },
        },
      },
    },
    responses: {
      201: {
        description: "Product created",
        example: { id: "p3", name: "Keyboard", price: 79.99 },
      },
      400: {
        description: "Invalid data",
        example: { error: "Invalid data" },
      },
    },
  })
  async create(@Req() req, @Res() res) {
    const { name, price } = req.body;
    if (!name || typeof price !== "number") {
      return res.status(400).json({ error: "Invalid data" });
    }
    const newProduct = { id: `p${Date.now()}`, name, price };
    this.products.push(newProduct);
    res.status(201).json(newProduct);
  }
}

const app = new AzuraClient();

setupSwaggerWithControllers(
  app,
  {
    title: "Products API",
    description: "Automatic documentation using decorators",
    version: "1.0.0",
    tags: [{ name: "Products", description: "Product management" }],
  },
  [ProductController]
);

await app.listen(3000);
```

**Advantages of `@Swagger`:**

- ✅ Everything in one place
- ✅ Cleaner and more readable
- ✅ Less boilerplate
- ✅ Full TypeScript support
- ✅ Inline examples

### Individual Decorators (Traditional API)

#### @ApiDoc()

```typescript
import { Controller, Get, Req, Res } from "azurajs/decorators";
import { ApiDoc } from "azurajs/swagger";

@Controller("/profile")
class ProfileController {
  @Get("/:id")
  @ApiDoc({
    summary: "Get profile by ID",
    description: "Retrieve public profile information",
  })
  async getProfile(@Req() req, @Res() res) {
    const profiles = {
      "1": { id: "1", displayName: "vinicius", bio: "Fullstack dev" },
    };
    const profile = profiles[req.params.id];
    if (!profile) return res.status(404).json({ error: "Not found" });
    res.json(profile);
  }
}
```

### @ApiResponse()

```typescript
import { Get } from "azurajs/decorators";
import { ApiResponse } from "azurajs/swagger";

class AccountController {
  @Get("/account")
  @ApiResponse(200, "User account", {
    example: { id: "1", email: "user@example.com", plan: "pro" },
  })
  @ApiResponse(401, "Unauthorized", { example: { error: "Unauthorized" } })
  async account(req, res) {
    res.json({ id: "1", email: "user@example.com", plan: "pro" });
  }
}
```

### @ApiParameter()

```typescript
import { Get } from "azurajs/decorators";
import { ApiParameter } from "azurajs/swagger";

class OrdersController {
  @Get("/:orderId")
  @ApiParameter("orderId", "path", {
    description: "Order ID",
    required: true,
    schema: { type: "string" },
    example: "ord_123",
  })
  async getOrder(req, res) {
    const orders = { ord_123: { id: "ord_123", total: 199.9 } };
    const order = orders[req.params.orderId];
    if (!order) return res.status(404).json({ error: "Order not found" });
    res.json(order);
  }
}
```

### @ApiBody()

```typescript
import { Post } from "azurajs/decorators";
import { ApiBody } from "azurajs/swagger";

class AuthController {
  @Post("/login")
  @ApiBody({
    description: "Login payload",
    required: true,
    example: { email: "user@example.com", password: "secret" },
  })
  async login(req, res) {
    const { email, password } = req.body;
    if (email === "user@example.com" && password === "secret") {
      return res.json({ token: "jwt.token.here" });
    }
    res.status(401).json({ error: "Invalid credentials" });
  }
}
```
