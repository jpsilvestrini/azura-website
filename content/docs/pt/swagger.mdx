import { Controller, Get, Post, Put, Delete, Req, Res } from "azurajs/decorators";
import { ApiDoc, ApiResponse, ApiParameter, ApiBody, ApiTags } from "azurajs/swagger";

@Controller("/customers")
@ApiTags("Customers")
class CustomersController {
  private customers = [{ id: "c1", name: "Empresa A", contact: "contato@empresa.com" }];

  @Get("/")
  @ApiDoc({ summary: "List all customers", description: "Retrieve a paginated list of customers" })
  @ApiParameter("page", "query", {
    description: "Page number",
    schema: { type: "integer", default: 1 },
  })
  @ApiParameter("limit", "query", {
    description: "Items per page",
    schema: { type: "integer", default: 10 },
  })
  @ApiResponse(200, "Success", {
    example: { customers: [{ id: "c1", name: "Empresa A" }], total: 1, page: 1 },
  })
  async list(@Req() req, @Res() res) {
    res.json({
      customers: this.customers,
      total: this.customers.length,
      page: req.query.page || 1,
    });
  }

  @Get("/:id")
  @ApiDoc({ summary: "Get customer by ID", description: "Retrieve detailed customer information" })
  @ApiParameter("id", "path", {
    description: "Customer ID",
    required: true,
    schema: { type: "string" },
  })
  @ApiResponse(200, "Customer found", {
    example: { id: "c1", name: "Empresa A", contact: "contato@empresa.com" },
  })
  @ApiResponse(404, "Customer not found")
  async get(@Req() req, @Res() res) {
    const customer = this.customers.find((c) => c.id === req.params.id);
    if (!customer) return res.status(404).json({ error: "Customer not found" });
    res.json(customer);
  }

  @Post("/")
  @ApiDoc({ summary: "Create new customer", description: "Create a new customer in the system" })
  @ApiBody({
    description: "Customer data",
    required: true,
    example: { name: "Empresa B", contact: "b@empresa.com" },
  })
  @ApiResponse(201, "Customer created", {
    example: { id: "c2", name: "Empresa B", contact: "b@empresa.com" },
  })
  @ApiResponse(400, "Invalid data")
  async create(@Req() req, @Res() res) {
    const { name, contact } = req.body;
    if (!name || !contact) return res.status(400).json({ error: "Invalid data" });
    const newCustomer = { id: `c${Date.now()}`, name, contact };
    this.customers.push(newCustomer);
    res.status(201).json(newCustomer);
  }

  @Put("/:id")
  @ApiDoc({ summary: "Update customer" })
  @ApiParameter("id", "path", {
    description: "Customer ID",
    required: true,
    schema: { type: "string" },
  })
  @ApiBody({ description: "Updated customer data", example: { name: "Empresa A Atualizada" } })
  @ApiResponse(200, "Customer updated")
  @ApiResponse(404, "Customer not found")
  async update(@Req() req, @Res() res) {
    const customer = this.customers.find((c) => c.id === req.params.id);
    if (!customer) return res.status(404).json({ error: "Customer not found" });
    Object.assign(customer, req.body);
    res.json(customer);
  }

  @Delete("/:id")
  @ApiDoc({ summary: "Delete customer" })
  @ApiParameter("id", "path", { description: "Customer ID", required: true })
  @ApiResponse(204, "Customer deleted")
  @ApiResponse(404, "Customer not found")
  async remove(@Req() req, @Res() res) {
    const idx = this.customers.findIndex((c) => c.id === req.params.id);
    if (idx === -1) return res.status(404).json({ error: "Customer not found" });
    this.customers.splice(idx, 1);
    res.status(204).send();
  }
}
